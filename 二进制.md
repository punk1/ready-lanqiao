# 转化二进制

```
#include <bits/stdc++.h>
#include<iostream>
using namespace std;
int arr[8];//用于存放转为二进制的数 
void exchange(int value)//转为二进制 
{
	int  i=0;
	int  va;
	va = fabs(value);
	while(va!=0)
	{
		arr[i]=va % 2;
		va/=2;
		i++;
	}
	
	//对负数的转化二进制的操作 
	if (value < 0)
	{
		for (int i = 0;i<=7;i++)
		{
			if (arr[i]  == 1 && i < 7)//从右边开始第一个 为1 的数之前的取反 
			{
				int j;
				for (j= i+1;j<= 7;j++)
				{
					if (arr[j]==0)
					{
						arr[j]=1;
					}
					else arr[j]=0;
				}
				break;
			}
		}
	}
}

int main()
{
	int i; 
	int a[32];
	for(int e= 0;e<= 9;e++)//10 行数据 
	{
		for (int i=0;i<=31;i++)
		{
			cin>>a[i];
		}
		int j;
		for (int i=0;i<=31;i++)
		{
			exchange(a[i]);
			for (j = 7;j>= 0;j--)
			{
				if (arr[j]==1)
				{
					printf("*");
				}
				else printf (" ");
			}
			int z;
			for (z=0;z<=7;z++)
			{
				arr[z]=0;
			}
			if (i%2==1)
			{
				printf("\n"); 
			} 
		}
	}
	return 0;
		
}
```
