#include <bits/stdc++.h>
#include<iostream>
using namespace std;

int main()
{
	long long int a,b,n;
	cin>>a>>b>>n;
	a=a%b;
	while(n-10>0)
	{
		a*=1e10;
		a%=b;
		n-=10;
	}

	for (int i =0;i<n+2;i++)
	{
		a=a*10;
		if (i>n-2)cout<<a/ b;
		a=a%b;
	}
	return 0;
	
}
